Citing GPkit
************

    @Misc{gpkit,
          author={Edward Burnell and Warren Hoburg},
          title={GPkit software for geometric programming},
          howpublished={\url{https://github.com/convexengineering/gpkit}},
          year={2018},
          note={Version 0.7.0}
         }
